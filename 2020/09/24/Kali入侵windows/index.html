命令 说明
sessions -i <id值> #进入会话 -k 杀死会话
background #将当前会话放置后台
getuid #查看已有的权限
getpid #获取当前进程的pid
sysinfo #查看目标系统信息
ps #查看当前活跃进程
kill #杀死进程值
reboot/shutdown #重启/关机
shell #进入目标主机cmd shell
uictl [enable/disable] [keyboard/mouse/all] #开启或禁用键盘/鼠标
uictl disable mouse #禁用鼠标
uictl disable keyboard #禁用键盘
